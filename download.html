Python

import json

# Load the JSON file
with open('medical_papers_major_medical_20250420.json', 'r', encoding='utf-8') as f:
    papers = json.load(f)

# Print the number of papers
print(f"Loaded {len(papers)} papers")

# Access the first paper
if papers:
    first_paper = papers[0]
    print(f"First paper title: {first_paper['title']}")
    print(f"Journal: {first_paper['journal']}")
    print(f"URL: {first_paper['url']}")

JavaScript

// Load the JSON file (in a browser)
fetch('medical_papers_major_medical_20250420.json')
  .then(response => response.json())
  .then(papers => {

    // Access the first paper

      const firstPaper = papers[0];
      console.log(`First paper title: ${firstPaper.title}`);
      console.log(`Journal: ${firstPaper.journal}`);
      console.log(`URL: ${firstPaper.url}`);
    }
  });
